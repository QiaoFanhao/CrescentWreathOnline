local CardBase = require("Base.CardBase")
local Card_23030 = CardBase.New(23030)

function Card_23030:OnPlay(context)
    context:AddSkillPoint(1)
    print("[Lua] +1 skill point, selecting enemy target...")

    local targetId = coroutine.yield(context:SelectPlayer(1))

    if targetId ~= -1 then
        print("[Lua] Target selected " .. targetId .. ", requesting 3 Magic damage...")

        local finalDmg = coroutine.yield(context:ApplyDamage(targetId, 3, "Magic"))

        if finalDmg > 0 then
            print("[Lua] Final damage dealt: " .. finalDmg)
            context:DealDamage(targetId, finalDmg, "Magic")
        else
            print("[Lua] Attack was fully blocked")
        end
    else
        print("[Lua] Target selection canceled")
    end
end

return Card_23030
