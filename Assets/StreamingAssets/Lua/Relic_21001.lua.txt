local Debug = CS.UnityEngine.Debug

-- 修改函数，使其在结束时断开引用
function OnPlay(cardData)
    Debug.Log("<color=cyan>[Lua]</color> 正在发动：" .. cardData.cardName)
    Debug.Log("<color=yellow>[效果]</color> 产出灵符：" .. cardData.coinBonus)
    
    -- 逻辑处理完成后，手动将局部引用设为 nil
    cardData = nil 
end
